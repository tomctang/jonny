def count_combinations_eigen(n):

    char_poly = r^4 - 2*r^3 + r^2 - 1  # eigenpolynomial r^4 - 2r^3 + r^2 - 1 =0  <=> recursion a(n) - 2a(n-1) + a(n-2) - a(n-4) =0
